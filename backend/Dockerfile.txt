# Build stage
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy csproj files and restore
COPY ["GGHub.WebAPI/GGHub.WebAPI.csproj", "GGHub.WebAPI/"]
COPY ["GGHub.Infrastructure/GGHub.Infrastructure.csproj", "GGHub.Infrastructure/"]
COPY ["GGHub.Application/GGHub.Application.csproj", "GGHub.Application/"]
COPY ["GGHub.Core/GGHub.Core.csproj", "GGHub.Core/"]
RUN dotnet restore "GGHub.WebAPI/GGHub.WebAPI.csproj"

# Copy everything else and build
COPY . .
WORKDIR "/src/GGHub.WebAPI"
RUN dotnet build "GGHub.WebAPI.csproj" -c Release -o /app/build

# Publish
FROM build AS publish
RUN dotnet publish "GGHub.WebAPI.csproj" -c Release -o /app/publish /p:UseAppHost=false

# Runtime stage
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS final
WORKDIR /app
EXPOSE 8080

COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "GGHub.WebAPI.dll"]
